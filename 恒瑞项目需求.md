-- 第一部分：key_id 不为空的数据
SELECT batch_id,
serial_no, monitoring_year, id, product_id, product_name, platform,
generic_name, url, specification, online_store_price, box_quantity,
unit_price_per_box, store_name, business_license_name, province, city,
sales_volume, category, created_time, crawled_time, repeated_count,
shipping_origin, key_id, name, address, standardized_province,
short_name, remarks
FROM haosen_project..hr_monitoring_data_tmp
WHERE key_id IS NOT NULL
    UNION ALL
    SELECT a.batch_id,
        a.serial_no, a.monitoring_year, a.id, a.product_id, a.product_name, 
        a.platform, a.generic_name, a.url, a.specification, a.online_store_price, 
        a.box_quantity, a.unit_price_per_box, a.store_name, a.business_license_name, 
        a.province, a.city, a.sales_volume, a.category, a.created_time, 
        a.crawled_time, a.repeated_count, a.shipping_origin,
        c.key_id, 
        c.name, 
        c.address,
        a.standardized_province, a.short_name, a.remarks
    FROM (
        SELECT batch_id,
            serial_no, monitoring_year, id, product_id, product_name, platform, 
            generic_name, url, specification, online_store_price, box_quantity, 
            unit_price_per_box, store_name, business_license_name, province, city, 
            sales_volume, category, created_time, crawled_time, repeated_count, 
            shipping_origin, standardized_province, short_name, remarks
        FROM haosen_project..hr_monitoring_data_tmp 
        WHERE key_id IS NULL
    ) a 
    LEFT JOIN (
        SELECT * FROM test..医院_标准名称
        UNION ALL
        SELECT * FROM test..医院_历史名称
        UNION ALL
        SELECT * FROM test..药店_标准名称
        UNION ALL
        SELECT * FROM test..药店_历史名称
    ) b ON a.business_license_name = REPLACE(REPLACE(REPLACE(b.name, '（', '('), '）', ')'), ' ', '') 
       AND a.standardized_province = b.province
    LEFT JOIN (
        SELECT keyid AS key_id, name AS name, address AS address 
        FROM HospitalInfo..hospital 
        UNION ALL
        SELECT key_id AS key_id, name AS name, address AS address 
        FROM HospitalInfo..company
    ) c ON b.keyid = c.key_id
) AS combined_data;
